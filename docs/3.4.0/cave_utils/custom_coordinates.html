  6    def __init__(self, length: float | int, width: float | int, height: float | int = 10000):
  7        """
  8        Creates a custom 2D or 3D Cartesian coordinate system with the origin (0, 0) located at the bottom-left
  9        of the map and x and y increasing in value whilst moving right and up along the flat plane, respectively. z
 10        increases whilst moving up along the vertical axis.
 11
 12        Arguments:
 13
 14        * **`length`**: `[float | int]` &rarr; The maximum x value of this coordinate system. Must be greater than 0.
 15        * **`width`**: `[float | int]` &rarr; The maximum y value of this coordinate system. Must be greater than 0.
 16        * **`height`**: `[float | int]` = `10000` &rarr; The maximum z value of this coordinate system. Must be greater than 0.
 17
 18        Returns:
 19
 20        * `[None]`
 21        """
 22        self.length = length
 23        self.width = width
 24        self.height = height
 25        self.radius = max(length, width) / (2 * math.pi)
 26        self.margin = abs(length - width) / 2
 27        self.__validate_coordinate_system__()
 28
 29    def convert_coordinates(self, coordinates: list):
 30        """
 31        Converts the coordinates in this coordinate system to a longitude-latitude-altitude system. Directly mutates the input.
 32
 33        Arguments:
 34
 35        * **`coordinates`**: `[list]` &rarr; A nested list of (x, y) or (x, y, z) coordinates. The level of nesting must be consistent throughout the list.
 36            * ** Examples **:
 37                * `[100,50]`
 38                * `[100,50,0]`
 39                * `[[100,50,0], [20,20,20]]`
 40                * `[[[100,50], [100,100]], [[0,0], [0,100]]]`
 41
 42        Returns:
 43
 44        * `[None]`
 45        """
 46        if isinstance(coordinates[0], (float, int)):
 47            self.convert_coordinate(coordinates)
 48        else:
 49            if not all(isinstance(sublist, list) for sublist in coordinates):
 50                raise ValueError("Input must be either a list of two or three numbers or lists.")
 51            for sublist in coordinates:
 52                self.convert_coordinates(sublist)

 54    def convert_coordinate(self, coordinate: list[float | int]):
 55        """
 56        Converts the given coordinate in this coordinate system to a longitude-latitude-altitude system. Directly mutates the input.
 57
 58        Arguments:
 59
 60        * **`coordinate`**: `list[float | int]` &rarr; The coordinate to be converted in this coordinate system in the format `[x,y,(optional z)]`.
 61
 62        Returns:
 63
 64        * `[None]`
 65        """
 66        self.__validate_list_coordinates__([coordinate])
 67        x = coordinate[0]
 68        y = coordinate[1] - self.width / 2
 69        if len(coordinate) == 3:
 70            z = coordinate[2]
 71            scale = 10000 / self.height
 72            altitude = z * scale
 73            coordinate[2] = altitude
 74
 75        if self.width > self.length:
 76            x += self.margin
 77
 78        longitude = (x / self.radius) * (180 / math.pi)
 79        latitude = (360 / math.pi) * (math.atan(math.exp(y / self.radius)) - math.pi / 4)
 80
 81        # Y values close to 0 will not display on map
 82        if latitude < -85.05 and coordinate[1] >= 0:
 83            latitude = -85.05
 84        coordinate[0] = longitude - 180
 85        coordinate[1] = latitude

 87    def serialize_coordinates(self, coordinates: list[list[float | int]]):
 88        """
 89        Serializes (x, y, z) coordinates in this coordinate system to a longitude-latitude-altitude system.
 90        Formula adapted from: https://en.wikipedia.org/wiki/Mercator_projection#Derivation
 91
 92        Arguments:
 93
 94        * **`coordinates`**: `list[list[float | int]]` &rarr; The coordinates to be converted in this coordinate system in the format `[[x1,y1,(optional z1)],[x2,y2,(optional z2)],...]`.
 95            * ** Example **: `[[0,0],[103.5,99.1],[76.55,350],[12.01,12.01]]`
 96            * ** Example with Altitude **: `[[0,0,0],[103.5,99.1,1],[76.55,350,0.2],[12.01,12.01,3.41]]`
 97            * ** Note **: All coordinates must have either no altitude or altitude values. There cannot be a mix of both.
 98
 99        Returns:
100
101        * `[list[list[float | int]]]` &rarr; The converted coordinates in the format `[[long1,lat1,(possible alt1)],[long2,lat2,(possible alt2)],...]`.
102        """
103        self.__validate_list_coordinates__(coordinates)
104        long_lat_coordinates = []
105
106        for coordinate in coordinates:
107            coordinate_copy = [coord for coord in coordinate]
108            self.convert_coordinate(coordinate_copy)
109            long_lat_coordinates.append(coordinate_copy)
110
111        return long_lat_coordinates

113    def serialize_nodes(self, coordinates: list[list[float | int]] | dict[str, list[float | int]]):
114        """
115        Serialize the given node coordinates in this coordinate system to a dictionary of the proper format to be used under `mapFeatures.data.*.data.location`.
116
117        Arguments:
118
119        * **`coordinates`**: `[list[list[float | int]] | dict[str, list[float | int]]]` &rarr; The coordinates to be serialized in this coordinate system in the format `[[x1,y1],[x2,y2],...]` or a dictionary with "x", "y", and an optional "z" key with lists of values for all coordinates.
120            * ** Example List Type **: `[[0,0],[103.5,99.1],[76.55,350],[12.01,12.01]]`
121            * ** Example Dictionary Type with Altitude **: `{"x": [0,103.5,76.55,12.01], "y": [0,99.1,350,12.01], "z": [0,1,0.2,3.41]}`
122
123        Returns:
124
125        * `[dict]` &rarr; The serialized location structure.
126        """
127        if isinstance(coordinates, list):
128            converted_coordinates = self.serialize_coordinates(coordinates)
129        elif isinstance(coordinates, dict):
130            self.__validate_dict_coordinates__(coordinates)
131            if "z" in coordinates:
132                list_coordinates = [
133                    list(coordinate_list)
134                    for coordinate_list in zip(coordinates["x"], coordinates["y"], coordinates["z"])
135                ]
136            else:
137                list_coordinates = [
138                    list(coordinate_list)
139                    for coordinate_list in zip(coordinates["x"], coordinates["y"])
140                ]
141            converted_coordinates = self.serialize_coordinates(list_coordinates)
142        else:
143            raise ValueError(
144                "Coordiates must be a list of coordinate values or a dictionary with 'x', 'y', and optional 'z' keys."
145            )
146
147        if len(converted_coordinates[0]) == 2:
148            return {
149                "latitude": [[coordinate[1]] for coordinate in converted_coordinates],
150                "longitude": [[coordinate[0]] for coordinate in converted_coordinates],
151            }
152        return {
153            "latitude": [[coordinate[1]] for coordinate in converted_coordinates],
154            "longitude": [[coordinate[0]] for coordinate in converted_coordinates],
155            "altitude": [[coordinate[2]] for coordinate in converted_coordinates],
156        }

242        Raises:

249    def __validate_list_coordinates__(self, coordinates: list[list[float | int]]):
250        """
251        Validates that the given coordinates are in the proper format.
252
253        Arguments:
254
255        * **`coordinates`**: `list[list[float | int]]` &rarr; The coordinates to be validated.
256
257        Raises:
258
259        * **`ValueError`** &rarr; If the coordinate data is not in the proper format.
260        """
261        if not (
262            all(len(coord) == 2 for coord in coordinates)
263            or all(len(coord) == 3 for coord in coordinates)
264        ):
265            raise ValueError("Coordinates must all have either two elements or three elements.")
266        for coordinate in coordinates:
267            if not (0 <= coordinate[0] <= self.length and 0 <= coordinate[1] <= self.width):
268                raise ValueError("The given x and y coordinates are out of range.")
269            if len(coordinate) == 3 and not (0 <= coordinate[2] <= self.height):
270                raise ValueError("The given z coordinates are out of range.")
271
272    def __validate_dict_coordinates__(self, coordinates: dict[str, list[float | int]]):
273        """
274        Validates that the given coordinates are in the proper format.
275
276        Arguments:
277
278        * **`coordinates`**: `dict[str, list[float | int]]` &rarr; The coordinates to be validated.
279
280        Raises:
281
282        * **`ValueError`** &rarr; If the coordinate data is not in the proper format.
283        """
284        if "x" not in coordinates or "y" not in coordinates:
285            raise ValueError("Coordinates must contain 'x' and 'y' keys.")
286        if len(coordinates["x"]) != len(coordinates["y"]):
287            raise ValueError("The number of x and y values must match.")
288        if not (
289            all(0 <= x <= self.length for x in coordinates["x"])
290            and all(0 <= y <= self.width for y in coordinates["y"])
291        ):
292            raise ValueError("The given x and y coordinates are out of range.")
293        if "z" in coordinates:
294            if len(coordinates["x"]) != len(coordinates["z"]):
295                raise ValueError("The number of z values must match x and y.")
296            if not all(0 <= z <= self.height for z in coordinates["z"]):
297                raise ValueError("The given z coordinates are out of range.")
