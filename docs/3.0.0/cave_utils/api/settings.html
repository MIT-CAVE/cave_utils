 91    def spec(
 92        showToolbar: bool = True,
 93        locale: str = "en-US",
 94        precision: int = 2,
 95        trailingZeros: bool = False,
 96        notation: [str, None] = None,
 97        notationDisplay: [str, None] = None,
 98        fallbackValue: [str, None] = "N/A",
 99        unit: [str, None] = None,
100        unitPlacement: [str, None] = None,
101        legendPrecision: [str, None] = None,
102        legendNotation: [str, None] = None,
103        legendNotationDisplay: [str, None] = None,
104        legendMinLabel: [str, None] = None,
105        legendMaxLabel: [str, None] = None,
106        **kwargs,
107    ):
108        """
109        Arguments:
110
111        * **`showToolbar`**: `[bool]` = `True` &rarr; If `True`, chart toolbars will be displayed by default.
112        * **`locale`**: `[str]` = `"en-US"` &rarr;
113            * Format numeric values based on language and regional conventions.
114            * **Note**: This attribute only applies to `"num"` props or `stats`.
115            * **See**: [Locale identifier][].
116        * **`precision`**: `[int]` = `2` &rarr; The number of decimal places to display.
117            * **Notes**:
118                * Set the precision to `0` to attach an integer constraint.
119                * This attribute only applies to `"num"` props or `stats`.
120        * **`trailingZeros`**: `[bool]` = `False` &rarr; If `True`, trailing zeros will be displayed.
121            * **Notes**:
122                * This ensures that all precision digits are shown. For example: `1.5` &rarr; `1.500` when precision is `3`.
123                * This attribute only applies to `"num"` props or `stats`.
124        * **`notation`**: `[int]` = `"standard"` &rarr; The formatting style of a numeric value.
125            * **Accepted Values**:
126                    * `"standard"`: Plain number formatting
127                    * `"compact"`: Resembles the [metric prefix][] system
128                    * `"scientific"`: [Scientific notation][]
129                    * `"engineering"`: [Engineering notation][]
130                    * `"precision"`: Emulates the [Number.prototype.toPrecision][] method
131            * **Notes**:
132                * If left unspecified (i.e. `None`), the prop's or stat's `notation` will be used or in case the latter is undefined, `settings.notation` will be used.
133                * This attribute only applies to `"num"` props or `stats`.
134        * **`notationDisplay`**: `[str]` = `"e+"` | `"short"` | `None` &rarr; Further customize the formatting within the selected `notation`.
135            * **Accepted Values**:
136                * When **`notation`** == `"compact"`:
137                    * `"short"`: Add symbols `K`, `M`, `B`, and `T` (in `"en-US"`) to denote thousands, millions, billions, and trillions, respectively.
138                    * `"long"`: Present numeric values with the informal suffix words `thousand`, `million`, `billion`, and `trillion` (in `"en-US"`).
139                * When **`notation`** == `"scientific"`, `"engineering"` or `"precision"`:
140                    * `"e"`: Exponent symbol in lowercase as per the chosen `locale` identifier
141                    * `"e+"`: Similar to `"e"`, but with a plus sign for positive exponents.
142                    * `"E"`: Exponent symbol in uppercase as per the chosen `locale` identifier
143                    * `"E+"`: Similar to `"E"`, but with a plus sign for positive exponents.
144                    * `"x10^"`: Formal scientific notation representation
145                    * `"x10^+"`: Similar to `"x10^"`, with a plus sign for positive exponents.
146                * When **`notation`** == `"standard"`:
147                    * No `notationDisplay` option is allowed for a `"standard"` notation
148            * **Notes**:
149                * No `notationDisplay` option is provided for a `"standard"` notation
150                * The options `"short"` and `"long"` are only provided for the `"compact"` notation
151                * The options `"e"`, `"e+"`, `"E"`, `"E+"`, `"x10^"`, and `"x10^+"` are provided for the `"scientific"`, `"engineering"` and `"precision"` notations
152                * If `None`, it defaults to `"short"` for `"compact"` notation, and to `"e+"` for `"scientific"`, `"engineering"`, or `"precision"` notations. For the `"standard"` option, the value remains `None`.
153                * This attribute only applies to `"num"` props or `stats`.
154        * **`fallbackValue`**: [str] = `"N/A"` &rarr; A value to show when a numeric value is missing or invalid.
155            * **Note**: This attribute only applies to `"num"` props or `stats`.
156        * **`unit`**: `[str]` = `None` &rarr; The unit to use for a prop or stat.
157            * **Note**: This attribute only applies to `"num"` props or `stats`.
158        * **`unitPlacement`**: `[str]` = `"afterWithSpace"` &rarr; The position of the `unit` symbol relative to a value.
159            * **Accepted Values**:
160                * `"after"`: The `unit` appears after the value.
161                * `"afterWithSpace"`: The `unit` appears after the value, separated by a space.
162                * `"before"`: The `unit` appears before the value.
163                * `"beforeWithSpace"`: The unit is placed before the value, with a space in between.
164            * **Note**: This attribute only applies to `"num"` props or `stats`.
165        * **`legendPrecision`**: `[int]` = `None` &rarr;
166            * The number of decimal places to display in the Map Legend.
167            * **Notes**:
168                * Set the precision to `0` to attach an integer constraint.
169                * If left unspecified (i.e. `None`), the prop's or stat's `precision` will be used or in case the latter is undefined, `settings.precision` will be used.
170                * This attribute only applies to `"num"` props or `stats`.
171        * **`legendNotation`**: `[int]` = `"standard"` &rarr; The formatting style of a numeric value.
172            * **Accepted Values**:
173                * `"standard"`: Plain number formatting
174                * `"compact"`: Resembles the [metric prefix][] system
175                * `"scientific"`: [Scientific notation][]
176                * `"engineering"`: [Engineering notation][]
177                * `"precision"`: Emulates the [Number.prototype.toPrecision][] method
178            * **Notes**:
179                * If left unspecified (i.e. `None`), the prop's or stat's `notation` will be used or in case the latter is undefined, `settings.notation` will be used.
180                * This attribute only applies to `"num"` props or `stats`.
181        * **`legendNotationDisplay`**: `[str]` = `"e+"` | `"short"` | `None` &rarr; Further customize the formatting within the selected `legendNotation`.
182            * **Accepted Values**:
183                * `"short"`: Add symbols `K`, `M`, `B`, and `T` (in `"en-US"`) to denote thousands, millions, billions, and trillions, respectively.
184                * `"long"`: Present numeric values with the informal suffix words `thousand`, `million`, `billion`, and `trillion` (in `"en-US"`).
185                * `"e"`: Exponent symbol in lowercase as per the chosen `locale` identifier
186                * `"e+"`: Similar to `"e"`, but with a plus sign for positive exponents.
187                * `"E"`: Exponent symbol in uppercase as per the chosen `locale` identifier
188                * `"E+"`: Similar to `"E"`, but with a plus sign for positive exponents.
189                * `"x10^"`: Formal scientific notation representation
190                * `"x10^+"`: Similar to `"x10^"`, with a plus sign for positive exponents.
191            * **Notes**:
192                * No `legendNotationDisplay` option is provided for a `"standard"` legend notation
193                * The options `"short"` and `"long"` are only provided for the `"compact"` legend notation
194                * The options `"e"`, `"e+"`, `"E"`, `"E+"`, `"x10^"`, and `"x10^+"` are provided for the `"scientific"`, `"engineering"` and `"precision"` legend notations
195                * If left unspecified (i.e. `None`), the prop's or stat's `notationDisplay` will be used or in case the latter is undefined, `settings.notationDisplay` will be used.
196                * This attribute only applies to `"num"` props or `stats`.
197        * **`legendMinLabel`**: `[str]` = `None` &rarr;
198            * A custom and descriptive label in the Map Legend used to identify the lowest data point.
199            * **Notes**:
200                * Takes precedence over other formatting, except when used in a node cluster and the `cave_utils.api.maps.group` attribute is `True`. In this case, the min value within the node cluster is displayed.
201                * This attribute only applies to `"num"` props or `stats`.
202        * **`legendMaxLabel`**: `[str]` = `None` &rarr;
203            * A custom and descriptive label in the Map Legend used to identify the highest data point.
204            * **Notes**:
205                * Takes precedence over other formatting, except when used in a node cluster and the `cave_utils.api.maps.group` attribute is `True`. In this case, the max value within the node cluster is displayed.
206                * This attribute only applies to `"num"` props or `stats`.
207
208        [locale identifier]: https://en.wikipedia.org/wiki/IETF_language_tag
209        [metric prefix]: https://en.wikipedia.org/wiki/Metric_prefix
210        [Scientific notation]: https://en.wikipedia.org/wiki/Scientific_notation
211        [Engineering notation]: https://en.wikipedia.org/wiki/Engineering_notation
212        [Number.prototype.toPrecision]: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Number/toPrecision
213        """
214        passed_values = {k: v for k, v in locals().items() if (v is not None) and k != "kwargs"}
215        required_fields = []
216        if notationDisplay:
217            required_fields += ["notation"]
218        if legendNotationDisplay:
219            required_fields += ["legendNotation"]
220        missing_required = pamda.difference(required_fields, list(passed_values.keys()))
221        if len(missing_required) > 0:
222            raise Exception(f"Missing required fields: {str(missing_required)}")
223
224        notationDisplay_options_dict = {
225            "compact": ["short", "long"],
226            "scientific": ["e", "e+", "E", "E+", "x10^", "x10^+"],
227            "engineering": ["e", "e+", "E", "E+", "x10^", "x10^+"],
228            "precision": ["e", "e+", "E", "E+", "x10^", "x10^+"],
229            "standard": [],
230        }
231        notation = passed_values.get("notation", "standard")
232        legendNotation = passed_values.get("legendNotation", "standard")
233        return {
234            "kwargs": kwargs,
235            "accepted_values": {
236                "unitPlacement": ["after", "afterWithSpace", "before", "beforeWithSpace"],
237                "notation": ["standard", "compact", "scientific", "engineering", "precision"],
238                "notationDisplay": notationDisplay_options_dict.get(notation, []),
239                "legendNotation": ["standard", "compact", "scientific", "engineering", "precision"],
240                "legendNotationDisplay": notationDisplay_options_dict.get(legendNotation, []),
241            },
242        }

264@type_enforced.Enforcer
265class settings_sync_star(ApiValidator):
266    """
267    The sync settings are located under the path **`settings.sync.*`**.
268
269    By default, the each client will automatically sync all data structures with the server. This behavior can be modified using this
270    API structure.
271
272
273    """
274
275    @staticmethod
276    def spec(name: str, showToggle: bool, value: bool, data: dict[list[str | int]], **kwargs):
277        """
278
279        Arguments:
280
281        * **`name`**: `[str]` &rarr; The name of the sync setting. Only for display purposes if `showToggle` is `True`.
282        * **`showToggle`**: `[bool]` &rarr; If `True`, the toggle will be displayed.
283        * **`value`**: `[bool]` &rarr; The initial value of the toggle.
284            * **Notes**:
285                * This value determines if the data structure is synced with the server.
286                * This value is only relevant if it is `False`, as the entire data structure is synced by default.
287        * **`data`**: `[dict[list[str|int]]]` &rarr; A set of paths that allow you to toggle syncing.
288            * **Notes**:
289                * This is a dict of lists of strings or integers.
290                * Each key is a custom key for your own organizational purposes.
291                * Each value is a list of items that make up a path to a data structure in the api.
292                    * Each item in the list is either a string or an integer.
293                * If the path is valid and the value is `False`, anything under that path will not be synced with the server.
294                    * Essentially, each client will have its own divergent copy of the data structure under that path.
295
296        **Example**:
297
298        Allow users to not sync which pane or modal is open across clients in the same session:
299        ```
300        "sync: {
301            "panes": {
302                "name": "Panes And Modals",
303                "showToggle": True,
304                "value": False,
305                "data": {
306                    "ab1": ["panes", "paneState"],
307                },
308            }
309        }
310        ```
311        * **Note**: In this example the `ab1` key is arbitrary. It is used for your own organizational purposes.
312        * **Note**: In this example, `["panes", "paneState"]` refers to the data structure that determines which pane or modal is open.
313            * By choosing not to sync this data structure, each client can have its own divergent launched pane and launched modal state.
314            * See: `cave_utils.api.panes.panes_paneState_star` for more information on this data structure.
315
316        """
317        return {"kwargs": kwargs, "accepted_values": {}}
318
319    def __extend_spec__(self, **kwargs):
320        root_data = kwargs.get("root_data", {})
321        for key, path in self.data.get("data", {}).items():
322            if not pamda.hasPath(path, root_data):
323                self.__warn__(f"Path {path} does not exist.", path=["data", key])

327class settings_time(ApiValidator):
328    """
329    The time settings are located under the path **`settings.time`**.
330    """
331
332    @staticmethod
333    def spec(timeLength: int, timeUnits: str, looping: bool, speed: [float, int], **kwargs):
334        """
335        Arguments:
336
337        * **`timeLength`**: `[int]` &rarr; The amount of time values to display.
338        * **`timeUnits`**: `[str]` &rarr; The units of time to display.
339            * **Example**: `"Decade"`.
340        * **`looping`**: `[bool]` &rarr; If `True`, the time animation will automatically restart from the beginning once it reaches the end.
341        * **`speed`**: `[float]` &rarr; The speed at which the animation advances to the next time step.
342            * **Note**: While `speed` is intended to be measured in frames per second (fps), performance may degrade as the size of the time-varying data increases, resulting in slower animation.
343        """
344        return {"kwargs": kwargs, "accepted_values": {}}
345
346    def __extend_spec__(self, **kwargs):
347        timeLength = self.data.get("timeLength")
348        if timeLength < 1:
349            self.__error__(f"Time length must be greater than 0.", path=["timeLength"])
350        speed = self.data.get("speed")
351        accepted_speed_values = {0.5, 0.75, 1, 1.25, 1.5, 2}
352        if speed not in accepted_speed_values:
353            self.__error__(
354                f"speed must be one of the following values: {accepted_speed_values}.",
355                path=["speed"],
356            )
